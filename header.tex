% Skills
\newcommand{\skill}[1]{
  \begin{flushleft}
    \textbf{
      \texttt{
        \LARGE{\color{Emerald} #1}
      }
    }
  \end{flushleft}
  \vspace{-5pt}
}

% Subskill
\newcommand{\subskill}[4]{
  \vspace{5pt}
  \begin{tabular}[t]{l@{}}
    \ifempty{#2}
    \textbf{#1}
    \else
    \textbf{#1} (#2)
    \fi
  \end{tabular}
  \hfill
  \begin{tabular}[t]{r@{}}
      \textbf{#3} #4
  \end{tabular}
}

% Overview (like subskill)

\newcommand{\overview}[2]{
  \vspace{3pt}
  \begin{tabular}[t]{l@{}}
    \textbf{#1}:
  \end{tabular}
  \hfill
  \begin{tabular}[t]{r@{}}
      #2
  \end{tabular}\\
}

% Custom item command
